BEGIN;
CREATE DATABASE IF NOT EXISTS Codtopia;
USE Codtopia;

CREATE TABLE IF NOT EXISTS GUNS (
	GUN_ID				NUMERIC(10) PRIMARY KEY,
    GUN_NAME			VARCHAR(50) NOT NULL UNIQUE,
    GUN_CATEGORY		VARCHAR(30) NOT NULL,
    GUN_UNLOCK_LEVEL	NUMERIC(5) 	NOT NULL
);

CREATE TABLE IF NOT EXISTS ATTACHMENTS (
	ATM_ID				NUMERIC(10) PRIMARY KEY,
    ATM_NAME			VARCHAR(50) NOT NULL UNIQUE,
    ATM_CATEGORY		VARCHAR(30) NOT NULL
);

CREATE TABLE IF NOT EXISTS EQUIPMENT (
	EQ_ID				NUMERIC(10) PRIMARY KEY,
    EQ_NAME				VARCHAR(30) NOT NULL UNIQUE,
    EQ_CATEGORY			VARCHAR(15) NOT NULL,
    EQ_UNLOCK_LEVEL 	NUMERIC(5) 	NOT NULL
);

CREATE TABLE IF NOT EXISTS FIELD_UPGRADES (
	FU_ID				NUMERIC(10) PRIMARY KEY,
    FU_NAME 			VARCHAR(30) NOT NULL UNIQUE,
    FU_UNLOCK_LEVEL 	NUMERIC(5) 	NOT NULL
);

CREATE TABLE IF NOT EXISTS PERKS (
	PERK_ID				NUMERIC(10) PRIMARY KEY,
    PERK_NAME			VARCHAR(20) NOT NULL UNIQUE,
    PERK_CATEGORY		VARCHAR(10) NOT NULL,
    PERK_UNLOCK_LEVEL	NUMERIC(5) 	NOT NULL
);

CREATE TABLE IF NOT EXISTS LOADOUT (
	LOAD_ID		INT PRIMARY KEY AUTO_INCREMENT,
    LOAD_NAME 	VARCHAR(50),
    CLASS_DESCRIPTION VARCHAR(100),
    GUN_NAME		VARCHAR(50),
    ATM1_NAME		VARCHAR(50),
    ATM2_NAME		VARCHAR(50),
    ATM3_NAME		VARCHAR(50),
    ATM4_NAME		VARCHAR(50),
    ATM5_NAME		VARCHAR(50),
    ATM6_NAME		VARCHAR(50),
    EQ1_NAME		VARCHAR(50),
    EQ2_NAME		VARCHAR(50),
    FU_NAME		VARCHAR(50),
    PERK1_NAME	VARCHAR(50),
    PERK2_NAME	VARCHAR(50),
    PERK3_NAME	VARCHAR(50)
);
